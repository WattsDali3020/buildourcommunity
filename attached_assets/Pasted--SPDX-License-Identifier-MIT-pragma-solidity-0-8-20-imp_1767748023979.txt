pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/token/ERC1155/extensions/ERC1155Supply.sol";
import "@openzeppelin/contracts/token/ERC1155/extensions/ERC1155Burnable.sol";
import "@openzeppelin/contracts/access/AccessControl.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/structs/EnumerableSet.sol";

contract PropertyToken is ERC1155, ERC1155Supply, ERC1155Burnable, AccessControl, Pausable, ReentrancyGuard {
    using EnumerableSet for EnumerableSet.UintSet;

    bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");
    bytes32 public constant PAUSER_ROLE = keccak256("PAUSER_ROLE");
    bytes32 public constant WHITELIST_ADMIN_ROLE = keccak256("WHITELIST_ADMIN_ROLE");
    bytes32 public constant PHASE_ADVANCER_ROLE = keccak256("PHASE_ADVANCER_ROLE");

    enum Phase { County, State, National, International }

    struct Property {
        string name;
        string uri;
        uint256 totalSupply;
        uint256 mintedSupply;
        uint256 fundingTarget;
        uint256 fundingDeadline;
        Phase currentPhase;
        bool isActive;
        bool isFunded;
    }

    uint256 public constant COUNTY_ALLOCATION = 4000;
    uint256 public constant STATE_ALLOCATION = 3000;
    uint256 public constant NATIONAL_ALLOCATION = 2000;
    uint256 public constant INTERNATIONAL_ALLOCATION = 1000;

    uint256 public constant COUNTY_MULTIPLIER = 10000;
    uint256 public constant STATE_MULTIPLIER = 15000;
    uint256 public constant NATIONAL_MULTIPLIER = 22500;
    uint256 public constant INTERNATIONAL_MULTIPLIER = 30000;

    uint256 public constant COUNTY_VOTING_MULTIPLIER = 15000;
    uint256 public constant STATE_VOTING_MULTIPLIER = 12500;
    uint256 public constant NATIONAL_VOTING_MULTIPLIER = 10000;
    uint256 public constant INTERNATIONAL_VOTING_MULTIPLIER = 7500;

    uint256 public basePrice = 12500000000000000; // ~$12.50 wei

    mapping(uint256 => Property) public properties;
    uint256 public propertyCount;

    mapping(uint256 => mapping(Phase => uint256)) public phaseAllocations;
    mapping(uint256 => mapping(Phase => uint256)) public phaseMinted;
    mapping(uint256 => mapping(address => mapping(Phase => uint256))) public holdingsByPhase;

    mapping(address => bool) public isWhitelisted;
    mapping(address => Phase) public addressPhaseEligibility;

    EnumerableSet.UintSet private _allProperties;

    event PropertyCreated(uint256 indexed propertyId, string name, uint256 totalSupply);
    event TokensMinted(uint256 indexed propertyId, address indexed to, uint256 amount, Phase phase);
    event TokensBurned(uint256 indexed propertyId, address indexed from, uint256 amount, Phase phase);
    event PhaseAdvanced(uint256 indexed propertyId, Phase newPhase, string reason);
    event PropertyFunded(uint256 indexed propertyId);
    event BasePriceUpdated(uint256 newPrice);

    constructor(string memory uri_) ERC1155(uri_) {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(MINTER_ROLE, msg.sender);
        _grantRole(PAUSER_ROLE, msg.sender);
        _grantRole(WHITELIST_ADMIN_ROLE, msg.sender);
        _grantRole(PHASE_ADVANCER_ROLE, msg.sender);
    }
